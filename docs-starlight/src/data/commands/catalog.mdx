    code: |
      terragrunt catalog --enable-shell
  - description: Enable hooks in templates when scaffolding from catalog.
    code: |
      terragrunt catalog --enable-hooks

  - catalog-enable-shell
  - catalog-enable-hooks

  - catalog-root-file-name
